### 引入class


``` python
class Value: # 定义名为Value的类

    def __init__(self,data): # 初始化，data记录的是Value(3.0)的3.0或者某个数字
        self.data = data # 记录data这个数据
               
    def __repr__(self): # __repr__是一种类中的魔术方法，使返回Value中的数据可视化
        return f"Value(data = {self.data})"
```


### 在类中创建加法和乘法，记录参与运算的数字和符号


``` python
class Value:
	# _student = () 有等号说明不是必要输入，()是指输入的内容是元组，目的是可传入参与运算的数据。孩子的爸爸妈妈。
	# _op的目的是传入运算符号。
    def __init__(self,data,_student = (),_op = '' ): ，
        self.data = data # 保存data
        self._prev = set(_student) # 保存孩子的爸妈是谁，set是去重
        self._op = _op # 保存运算符号
        self.grad = 0.0 # 设置初始数据的导数（梯度）=0.0
        
    def __repr__(self):
        return f"Value(data = {self.data})"

    def __add__(self,other):
    # 使用+之后，输出会自动加入Value这个类中，并初始化，self.data + other.data就是初始话中的data。
    # (self,other)对应是初始化中的_student = ()，输出结果out的爸爸妈妈。
    # '+'对应_op,指的是运算的符号为+号
        out = Value(self.data + other.data,(self,other),'+')
        return out # 返回输出值，记录
	
	# 与__mul__类似，这里代表*--乘法
    def __mul__(self,other):
        out = Value(self.data * other.data,(self,other),'*')
        return out
```

### 在python中运用链式法则

```  python
# 链式法则是导数中的内容，我们就用测试中的例子。
e = a*b
d = e + c

# d对d的导数是什么？ -->1.0
# d对e的导数？-->1.0
# d对c的导数？-->1.0
# d对a的导数？-->1.0*b
# d对b的导数？-->1.0*a

# 在python中算出d的结果后，如何计算d对e和c的导数，d对a和b的导数呢？
# 此时_prev=set(student)和_op派上用场，_prev是参与计算的数据,_op是参与计算的符号。有了这两样东西，就能进行求导了。

class Value:
    def __init__(self, data, _children=(), _op=''):
        self.data = data
        self.grad = 0.0
        self._prev = set(_children)
        self._backward = lambda: None # 默认什么都不做，没有这行，初始的数据会报错，
        # 因为他们没有父母，石头里蹦出来的

    def __add__(self, other):
		other = other if isinstance(other, Value) else Value(other)
        out = Value(self.data + other.data, (self, other), '+')
        
        # 定义加法的“责任分配”逻辑
        def _backward(): # （）里没有内容代表着不需要额外输入信息输入
            # 加法很公平：父母的 grad 增加量 = 孩子的 grad
            self.grad += out.grad # 为什么是+=？见例2。
            other.grad += out.grad
        
        # 保存定义的函数_backward到out里，在输入out._backward时才会运行
        out._backward = _backward 
        
        return out

    def __mul__(self, other):
		
		#  当写3.0*x时，3.0是一个float ，它没有.data，无法计算，使用以下函数来解决问题
		#特定语法--> isinstance(谁，属于哪一类)
	    other = other if isinstance(other, Value) else Value(other)
	    
        out = Value(self.data * other.data, (self, other), '*')
        
        # 定义乘法的“责任分配”逻辑（这就有点数学魔法了）
        def _backward():
            # 乘法有“置换效应”：
            # 左边盒子的影响力 = 右边的数值 * 孩子的 grad
            self.grad += other.data * out.grad
            other.grad += self.data * out.grad
        out._backward = _backward
        
        return out
        
    def __rmul__(self, other): # 解决数字*Value的问题。当python发现*左边是数字时，会调用数字的乘法，不知道怎么乘Value。使用这个魔法方法会自动转换成左边字母右边数字。
	    return self.__mul__(other)
        
        return out
```

<u>例2</u>

 $y = x^2$
 $z = x \times 3$
 $L = y + z$
 
根据微积分，$\frac{dL}{dx} = \frac{dy}{dx} + \frac{dz}{dx} = 2x + 3$。如果 $x=2$，那么梯度应该是 $2(2) + 3 = 7$。

``` python

x = Value(2.0)
y = x * x          # y = x^2 = 4.0
z = x * 3.0        # z = 3x  = 6.0
L = y + z          # L = 4 + 6 = 10.0

L.grad = 1.0

L._backward()     
y._backward()     
z._backward()      

# dL / dx = ???
# dL /dx = (dL / dy) * (dy / dx) + (dL/dz) * (dz / dx)
# x = 2 -->dL / dx = 1.0 * 2.0*2.0 + 1.0 * 3.0 =7.0  正确答案

# 没有+=时-->第二次计算的x的导数会覆盖第一次的导数
x.grad # 输出为3.0，
```
