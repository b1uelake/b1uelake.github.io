预备知识：魔术方法__xx__,是类中自带的，每个魔术方法都有特定的作用和触发条件。
例子：
`__init__`-->**初始化**：对象出生时自动调用。创建对象时自动运行。
`__repr__`-->**显示**：定义对象展示给开发者看的样子。打印变量时自动运行。
`__add__`-->**显示**：**加法**：让对象支持 `+` 号运算。有 + 号时自动运行。
`__mul__`-->**显示**：**乘法**：让对象支持 `*` 号运算。有 * 号时自动运行。

```python
# 创建一个类，名为Value
class Value:
# 括号中不带=是必须填的，带=是非必要填的，括号里没有而底下有的，是默认的不让初始时改变的。
	def __init__(self, data, _children=(), _op='', label=''):
	    self.data = data
	    self.grad = 0.0
	    self._backward = lambda: None # 对叶子节点定义一个空函数，防止求导时报错
	    self._prev = set(_children) # 这是表示_children的前序节点
	    self._op = _op
	    self.label = label

# 为了让输出的值可见，有没有的区别见下面图片
	def __repr__(self):
	    return f"Value(data={self.data})"
    
	def __add__(self, other):
	    out = Value(self.data + other.data, (self, other), '+') # out是局部变量
    
	    def _backward():
	      self.grad += 1.0 * out.grad # ‘+’的导数是1，out.grad是前面传来的梯度
	      other.grad += 1.0 * out.grad
	    out._backward = _backward  # 把右边赋值给左边
    
    return out # out返回的值包含_backward 

	  def __mul__(self, other):
	    out = Value(self.data * other.data, (self, other), '*')
    
	    def _backward():
	      self.grad += other.data * out.grad
	      other.grad += self.data * out.grad
	    out._backward = _backward
      
	    return out
  
	  # 定义一个双曲正切函数，讲结果锁在-1到1之间，有助于神经网络更快地收敛
	  def tanh(self):
	    x = self.data
	    t = (math.exp(2*x) - 1)/(math.exp(2*x) + 1)
	    out = Value(t, (self, ), 'tanh')
	    
	    def _backward():
	      self.grad += (1 - t**2) * out.grad
	    out._backward = _backward
	    
	    return out
  
	# 画一个拓扑图直观的理解。
	  def backward(self):
	    
	    topo = [] # 创建一个空的列表
	    visited = set() # set的作用是去重，输出去重后的集合
	    def build_topo(v):
	      if v not in visited:
	        visited.add(v)
	        for child in v._prev:
	          build_topo(child)
	        topo.append(v)
	    build_topo(self)
	    
	    self.grad = 1.0
	    for node in reversed(topo):
	      node._backward()

a = Value(2.0, label='a')
b = Value(-3.0, label='b')
c = Value(10.0, label='c')
e = a*b; e.label = 'e'
d = e + c; d.label = 'd'
f = Value(-2.0, label='f')
L = d * f; L.label = 'L'
```
